### 1.2 数据流

```
用户点击"搜索公众号"
      │
      ▼
Vue Component (AccountSearch)
      │
      │ emit('search', keyword)
      │
      ▼
useWeChatApi().searchBiz(keyword)
      │
      │ window.electronAPI.invoke('biz:search', keyword)
      │
      ▼
IPC: 'biz:search'
      │
      ▼
Main Process Handler
      │
      ├─► Database (查询账号 Cookie)
      │         │
      │         ▼
      ├─► axios.get('mp.weixin.qq.com/cgi-bin/searchbiz')
      │         │
      │         ▼
      ├─► 解析响应
      │         │
      │         ▼
      └─► 返回结果给 Renderer
              │
              ▼
Vue Component (更新 UI)
```

## 4. IPC 通信设计

### 4.1 预加载脚本

```typescript
// preload.ts
import { contextBridge, ipcRenderer } from 'electron';

contextBridge.exposeInMainWorld('electronAPI', {
  // 登录
  getQrcode: () => ipcRenderer.invoke('login:getQrcode'),
  checkScan: (uuid: string) => ipcRenderer.invoke('login:checkScan', uuid),
  confirmLogin: (cookie: string) => ipcRenderer.invoke('login:confirm', cookie),

  // 公众号
  searchBiz: (keyword: string) => ipcRenderer.invoke('biz:search', keyword),

  // 文章
  fetchArticles: (fakeid: string) => ipcRenderer.invoke('articles:fetch', fakeid),
  getArticleDetail: (link: string) => ipcRenderer.invoke('articles:detail', link),

  // 导出
  exportEpub: (articleIds: number[], options: object) =>
    ipcRenderer.invoke('export:epub', articleIds, options),
  selectSavePath: () => ipcRenderer.invoke('dialog:selectPath'),

  // 数据库
  dbQuery: (sql: string, params: any[]) => ipcRenderer.invoke('db:query', sql, params),
  dbRun: (sql: string, params: any[]) => ipcRenderer.invoke('db:run', sql, params),

  // 窗口控制
  minimize: () => ipcRenderer.invoke('window:minimize'),
  close: () => ipcRenderer.invoke('window:close'),
  startDrag: () => ipcRenderer.invoke('window:startDrag'),

  // 托盘
  hideToTray: () => ipcRenderer.invoke('tray:hide'),

  // 事件监听
  onLoginScanned: (callback: () => void) => {
    ipcRenderer.on('login:scanned', callback);
  },
  onLoginConfirmed: (callback: () => void) => {
    ipcRenderer.on('login:confirmed', callback);
  },
  onCollectProgress: (callback: (progress: number) => void) => {
    ipcRenderer.on('collect:progress', (_, progress) => callback(progress));
  }
});
```

### 4.2 主进程处理器

```typescript
// main.ts - IPC 处理器示例
import { ipcMain, dialog, BrowserWindow } from 'electron';
import axios from 'axios';
import { initDatabase, query, run } from './database/db';

// 获取登录二维码
ipcMain.handle('login:getQrcode', async () => {
  const response = await axios.get('https://mp.weixin.qq.com/cgi-bin/loginqrcode', {
    params: { action: 'getqrcode', random: Math.random() }
  });
  return response.data;
});

// 搜索公众号
ipcMain.handle('biz:search', async (_, keyword: string) => {
  // 获取保存的 cookie
  const account = await query('SELECT * FROM accounts LIMIT 1');

  const response = await axios.get('https://mp.weixin.qq.com/cgi-bin/searchbiz', {
    params: {
      action: 'search_biz',
      fakeid: '',
      keyword,
      token: account?.token,
      lang: 'zh_CN',
      f: 'json',
      ajax: 1
    },
    headers: { Cookie: account?.cookie }
  });

  return response.data;
});

// 选择保存路径
ipcMain.handle('dialog:selectPath', async () => {
  const result = await dialog.showSaveDialog({
    title: '保存电子书',
    defaultPath: '微信文章精选.epub',
    filters: [{ name: 'EPUB', extensions: ['epub'] }]
  });
  return result.filePath;
});
```

CREATE TABLE IF NOT EXISTS accounts (

### 6.2 useCollector (Composables)

```typescript
// renderer/composables/useCollector.ts
import { ref } from 'vue';

export function useCollector() {
  const isCollecting = ref(false);
  const progress = ref({ current: 0, total: 0, currentTitle: '' });
  const articles = ref([]);

  // 开始采集
  async function startCollect(fakeid: string) {
    isCollecting.value = true;
    progress.value = { current: 0, total: 0, currentTitle: '' };

    // 监听进度更新
    window.electronAPI.onCollectProgress((data) => {
      progress.value = data;
    });

    try {
      // 获取文章列表
      const list = await window.electronAPI.fetchArticles(fakeid);
      progress.value.total = list.length;

      // 获取每篇文章详情
      for (let i = 0; i < list.length; i++) {
        progress.value.current = i + 1;
        progress.value.currentTitle = list[i].title;

        const detail = await window.electronAPI.getArticleDetail(list[i].link);
        articles.value.push(detail);
      }

      return articles.value;
    } finally {
      isCollecting.value = false;
    }
  }

  return {
    isCollecting,
    progress,
    articles,
    startCollect
  };
}
```

### 6.3 useExporter (Composables)

```typescript
// renderer/composables/useExporter.ts
import { ref } from 'vue';

export function useExporter() {
  const isExporting = ref(false);
  const exportProgress = ref(0);

  // 导出 EPUB
  async function exportEpub(articleIds: number[], options: {
    title: string;
    author: string;
    savePath: string;
  }) {
    isExporting.value = true;
    exportProgress.value = 0;

    try {
      const filePath = await window.electronAPI.exportEpub(articleIds, options);
      return filePath;
    } finally {
      isExporting.value = false;
    }
  }

  // 选择保存路径
  async function selectSavePath() {
    return window.electronAPI.selectSavePath();
  }

  return {
    isExporting,
    exportProgress,
    exportEpub,
    selectSavePath
  };
}
```

## 7. 界面交互流程

### 7.1 步骤管理

```vue
<!-- App.vue -->
<template>
  <div class="app">
    <TitleBar />

    <div class="content">
      <StepIndicator :current="currentStep" :total="3" />

      <div class="step-content">
        <LoginStep v-if="currentStep === 1" @next="nextStep" />
        <CollectStep v-if="currentStep === 2" @next="nextStep" @prev="prevStep" />
        <ExportStep v-if="currentStep === 3" @prev="prevStep" @done="finish" />
      </div>
    </div>
  </div>
</template>

<script setup>
import { ref } from 'vue';
import TitleBar from './components/TitleBar.vue';
import StepIndicator from './components/StepIndicator.vue';
import LoginStep from './components/steps/LoginStep.vue';
import CollectStep from './components/steps/CollectStep.vue';
import ExportStep from './components/steps/ExportStep.vue';

const currentStep = ref(1);

function nextStep() {
  currentStep.value++;
}

function prevStep() {
  currentStep.value--;
}

function finish() {
  currentStep.value = 1; // 重置
}
</script>
```
