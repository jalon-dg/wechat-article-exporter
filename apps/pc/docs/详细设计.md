```sql
CREATE TABLE accounts (

CREATE TABLE articles (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    account_id INTEGER NOT NULL,
    fakeid TEXT NOT NULL,              -- 公众号 FakeID
    title TEXT NOT NULL,               -- 文章标题
    link TEXT NOT NULL UNIQUE,         -- 文章链接 (唯一)

    author TEXT,                       -- 作者
    source_url TEXT,                   -- 原文链接

    publish_date DATETIME,             -- 发布日期

    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
```

### 5.1 useWeChatApi (Composables)

```typescript
// useWeChatApi.ts
export function useWeChatApi() {
  const baseUrl = 'http://localhost:38765/api';

  // 获取登录二维码
  async function getQrcode(): Promise<QrcodeResult> {
    return $fetch(`${baseUrl}/login/qrcode`);
  }

  // 检查扫码状态
  async function checkScan(uuid: string): Promise<ScanResult> {
    return $fetch(`${baseUrl}/login/check`, { query: { uuid } });
  }

  // 确认登录
  async function confirmLogin(cookie: string): Promise<LoginResult> {
    return $fetch(`${baseUrl}/login/confirm`, {
      method: 'POST',
      body: { cookie }
    });
  }

  // 搜索公众号
  async function searchBiz(keyword: string): Promise<BizSearchResult> {
    return $fetch(`${baseUrl}/biz/search`, { query: { keyword } });
  }

  // 获取公众号文章列表
  async function getArticles(fakeid: string, begin = 0): Promise<ArticlesResult> {
    return $fetch(`${baseUrl}/biz/articles`, { query: { fakeid, begin } });
  }

  // 获取文章详情
  async function getArticleDetail(link: string): Promise<ArticleDetail> {
    return $fetch(`${baseUrl}/articles/detail`, { query: { link } });
  }

  return {
    getQrcode,
    checkScan,
    confirmLogin,
    searchBiz,
    getArticles,
    getArticleDetail
  };
}
```

### 5.2 useCollector (Composables)

```typescript
// useCollector.ts
export function useCollector() {
  const progress = ref<CollectProgress>({
    status: 'idle',
    total: 0,
    current: 0,
    currentArticle: '',
    percentage: 0
  });

  const isRunning = computed(() => progress.value.status === 'running');
  const isPaused = computed(() => progress.value.status === 'paused');

  // 开始采集
  async function startCollect(accountId: number, fakeid: string) {
    // 1. 获取文章列表
    const list = await fetchArticleList(fakeid);

    // 2. 更新进度
    progress.value.total = list.length;
    progress.value.status = 'running';

    // 3. 并发下载
    const queue = new PQueue({ concurrency: 3 });

    for (const article of list) {
      await queue.add(async () => {
        if (progress.value.status === 'paused') {
          await new Promise(r => setTimeout(r, 1000));
        }

        progress.value.currentArticle = article.title;
        progress.value.current++;
        progress.value.percentage = Math.round(
          (progress.value.current / progress.value.total) * 100
        );

        // 下载文章内容
        await downloadArticle(article.link);

        // 保存到数据库
        await saveArticle(article);
      });
    }

    progress.value.status = 'completed';
  }

  // 暂停采集
  function pause() {
    progress.value.status = 'paused';
  }

  // 继续采集
  function resume() {
    progress.value.status = 'running';
  }

  return {
    progress,
    isRunning,
    isPaused,
    startCollect,
    pause,
    resume
  };
}
```

```typescript
export function useExporter() {
  const isExporting = ref(false);
  const exportProgress = ref(0);

  // 导出 EPUB
  async function exportEpub(articleIds: number[], options: EpubOptions) {
    isExporting.value = true;
    exportProgress.value = 0;

    try {
      // 1. 获取文章内容
      const articles = await getArticlesByIds(articleIds);

      // 2. 生成 EPUB
      const epub = new EpubGenerator({
        title: options.title,
        author: options.author,
        cover: options.cover,
        content: articles.map(a => ({
          title: a.title,
          content: a.content,
          date: a.publish_date
        }))
      });

      // 3. 生成文件
      const buffer = await epub.generate((p) => {
        exportProgress.value = p;
      });

      // 4. 保存文件
      const filePath = await saveFile(buffer, `${options.title}.epub`);

      return { success: true, filePath };
    } finally {
      isExporting.value = false;
    }
  }

    // 类似 EPUB 实现
  }

  // 导出 DOCX
  async function exportDocx(articleIds: number[], options: DocxOptions) {
    // 类似 EPUB 实现
  }

  return {
    isExporting,
    exportProgress,
    exportEpub,
    exportPdf,
    exportDocx
  };
}
```
